172     /**
173      * Called when the Maven plug-in is executing. It creates an in-memory ZIP file of all the Chrome Extension
174      * source files, generates as signature using the private key from the PEM file, outputs a CRX file containing
175      * a header, the public key, the signature and the ZIP data.
176      *
177      * @throws MojoExecutionException If there was an error that should stop the build.
178      * @throws MojoFailureException   If there was an error but the build might be allowed to continue.
179      */
180     @Override
181     public final void execute() throws MojoExecutionException, MojoFailureException {
182 
183         // Make sure we have a manifest file for the CRX
184 
185         final File manifestFile = new File(crxSourceDirectory, "manifest.json");
186         if (!manifestFile.exists()) {
187             throw new MojoExecutionException("Missing manifest.json file");
188         }
189 
190         // Generate CRX file name
191 
192         final StringBuilder crxFilename = new StringBuilder();
193         crxFilename.append(finalName);
194         if (StringUtils.isNotEmpty(classifier)) {
195             crxFilename.append('-');
196             crxFilename.append(classifier);
197         }
198         final File crxDirectory = new File(outputDirectory, crxFilename.toString());
199         crxFilename.append(".crx");
200 
201         copyFiles(crxSourceDirectory, crxDirectory);
202 
203         // Generate the CRX file
204 
205         final File crxFile = new File(outputDirectory, crxFilename.toString());
206         final String[] includes = ParameterUtils.splitParameter(packagingIncludes);
207         final String[] excludes = ParameterUtils.splitParameter(packagingExcludes);
208 
209         crxArchiver.setPemFile(pemFile);
210         crxArchiver.setPemPassword(pemPassword);
211         crxArchiver.addDirectory(crxDirectory, includes, excludes);
212         crxArchiver.setDestFile(crxFile);
213 
214         try {
215             crxArchiver.createArchive();
216         } catch (final IOException e) {
217             throw new MojoExecutionException("Failed to package and sign the Google Chrome Extension", e);
218         } catch (final ArchiverException e) {
219             throw new MojoExecutionException(e.getMessage(), e);
220         }
221 
222         // Attach the artifact to the build life-cycle
223 
224         if (StringUtils.isNotEmpty(classifier)) {
225             projectHelper.attachArtifact(project, "crx", classifier, crxFile);
226         } else {
227             project.getArtifact().setFile(crxFile);
228         }
229     }

231     /**
232      * Recursively copy from a source directory to a destination directory applying resource filtering if necessary.
233      *
234      * @param source      The source directory.
235      * @param destination The destination directory.
236      * @throws MojoExecutionException If there was an error during the recursive copying or filtering.
237      * @since 1.2.0
238      */
239     private void copyFiles(final File source, final File destination) throws MojoExecutionException {
240         try {
241             if (!destination.exists() && !destination.mkdirs()) {
242                 throw new MojoExecutionException("Could not create directory: " + destination.getAbsolutePath());
243             }
244             for (final File sourceItem : source.listFiles()) {
245                 final File destinationItem = new File(destination, sourceItem.getName());
246                 if (sourceItem.isDirectory()) {
247                     copyFiles(sourceItem, destinationItem);
248                 } else {
249                     if (filtering && !isNonFilteredExtension(sourceItem.getName())) {
250                         mavenFileFilter.copyFile(sourceItem, destinationItem, true, getFilterWrappers(), null);
251                     } else {
252                         FileUtils.copyFile(sourceItem, destinationItem);
253                     }
254                 }
255             }
256         } catch (final MavenFilteringException e) {
257             throw new MojoExecutionException("Failed to build filtering wrappers", e);
258         } catch (final IOException e) {
259             throw new MojoExecutionException("Error copying file: " + source.getAbsolutePath(), e);
260         }
261     }

263     /**
264      * Determine whether the file name should be filtered or not based on the list of excluded file extensions.
265      *
266      * @param fileName The file name.
267      * @return {@code true} if the file extension is not excluded and the file should be filtered. Otherwise {@code
268      *         false}.
269      * @throws MojoExecutionException If there was an error determining whether the file name should be filtered.
270      * @since 1.2.0
271      */
272     private boolean isNonFilteredExtension(final String fileName) throws MojoExecutionException {
273         return !mavenResourcesFiltering.filteredFileExtension(fileName, nonFilteredFileExtensions);
274     }

276     /**
277      * Build a list of filter wrappers.
278      *
279      * @return The list of filter wrappers.
280      * @throws MojoExecutionException If there was a problem building the list of filter wrappers.
281      * @since 1.2.0
282      */
283     private List getFilterWrappers()
284             throws MojoExecutionException {
285         if (filterWrappers == null) {
286             try {
287                 final MavenResourcesExecution mavenResourcesExecution = new MavenResourcesExecution();
288                 mavenResourcesExecution.setEscapeString("\\");
289                 filterWrappers = mavenFileFilter.getDefaultFilterWrappers(project, filters, true, session,
290                         mavenResourcesExecution);
291             } catch (final MavenFilteringException e) {
292                 throw new MojoExecutionException("Failed to build filtering wrappers: " + e.getMessage(), e);
293             }
294         }
295         return filterWrappers;
296     }
