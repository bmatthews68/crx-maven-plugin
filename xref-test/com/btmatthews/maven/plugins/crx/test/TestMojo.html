53      @Mock

59      @Mock

65      @Mock
66      private MavenProjectHelper projectHelper;

68      /**
69       * Create the test fixtures including the CRX mojo and a mocked the Maven project model and Maven project
70       * helper.
71       *
72       * @throws Exception If there was an error configuring the CRX mojo.
73       */
74      protected void setUp() throws Exception {
75          initMocks(this);
76          mojo = new CRXMojo();
77          when(project.getArtifact()).thenReturn(artifact);
78          setVariableValueToObject(mojo, "outputDirectory", new File("target"));
79          setVariableValueToObject(mojo, "project", project);
80          setVariableValueToObject(mojo, "projectHelper", projectHelper);
81      }

83      /**
84       * Verify that a .crx file can be created without a pemPassword or classifier property.
85       *
86       * @throws Exception If there was an error executing the unit test.
87       */
88      public void testMojo() throws Exception {
89          setVariableValueToObject(mojo, "finalName", "HelloWorld");
90          setVariableValueToObject(mojo, "pemFile", new File("target/test-classes/crxtest.pem"));
91          setVariableValueToObject(mojo, "crxSourceDirectory", new File("target/test-classes/HelloWorld"));
92          mojo.execute();
93          verify(artifact).setFile(any(File.class));
94      }
95  
96      /**
97       * Verify that a .crx file can be created with a pemPassword but without a classifier property.
98       *
99       * @throws Exception If there was an error executing the unit test.
100      */
101     public void testMojoWithPasssword() throws Exception {
102         setVariableValueToObject(mojo, "finalName", "HelloWorld");
103         setVariableValueToObject(mojo, "pemFile", new File("target/test-classes/crxtest1.pem"));
104         setVariableValueToObject(mojo, "pemPassword", "everclear");
105         setVariableValueToObject(mojo, "crxSourceDirectory", new File("target/test-classes/HelloWorld"));
106         mojo.execute();
107         verify(artifact).setFile(any(File.class));
108     }

110     /**
111      * Verify that an exception is raised when trying to sign a .crx file with a nonexistent PEM file.
112      *
113      * @throws Exception If there was an error executing the unit test.
114      */
115     public void testMojoWhenPEMFileDoesNotExist() throws Exception {
116         try {
117             setVariableValueToObject(mojo, "finalName", "HelloWorld");
118             setVariableValueToObject(mojo, "pemFile", new File("target/test-classes/crxtest2.pem"));
119             setVariableValueToObject(mojo, "pemPassword", "everclear");
120             setVariableValueToObject(mojo, "crxSourceDirectory", new File("target/test-classes/HelloWorld"));
121             mojo.execute();
122             fail();
123         } catch (final MojoExecutionException e) {
124             assertEquals("Could not load the public/private key from the PEM file", e.getMessage());
125         }
126     }
