56      @Before
57      public void setUp() throws Exception {

59          signatureHelper = new CRXSignatureHelper();
60          keyFactory = KeyFactory.getInstance("RSA", "BC");
61      }
62  
63      /**
64       * Verify the {@link SignatureHelper#sign(byte[], java.security.PrivateKey)} method.
65       *
66       *                                                                                                               /(
67       * @throws Exception If there was an unexpected problem.
68       */
69      @Test
70      public void testSign() throws Exception {
71          final PKCS8EncodedKeySpec keySpec = new PKCS8EncodedKeySpec(DummyArchive.PRIVATE_KEY);
72          final PrivateKey privateKey = keyFactory.generatePrivate(keySpec);
73          final byte[] signature = signatureHelper.sign(DummyArchive.DATA, privateKey);
74          assertArrayEquals(DummyArchive.SIGNATURE, signature);
75      }
76  
77  
78      /**
79       * Verify the {@link SignatureHelper#check(byte[], java.security.PublicKey, byte[])}  method.
80       *
81       * @throws Exception If there was an unexpected problem.
82       */
83      @Test
84      public void testCheck() throws Exception {
85          final X509EncodedKeySpec keySpec = new X509EncodedKeySpec(DummyArchive.PUBLIC_KEY);
86          final PublicKey publicKey = keyFactory.generatePublic(keySpec);
87          final boolean result = signatureHelper.check(DummyArchive.DATA, publicKey, DummyArchive.SIGNATURE);
88          assertTrue(result);
89      }
