49       * Prepare for test case execution.

52      public void setUp() {
53          archiveHelper = new CRXArchiveHelper();
54      }
55  
56      /**
57       * Verify writing an archive and reading it back into memory.
58       *
59       * @throws Exception If there is an unexpected problem.
60       */
61      @Test
62      public void testReadWrite() throws Exception {
63          final File crxFile = new File(outputDirectory.getRoot(), "HelloWorld-1.0.0-SNAPSHOT.crx");
64          final CRXArchive crxOutArchive = new DummyArchive();
65          archiveHelper.writeArchive(crxFile, crxOutArchive);
66          final CRXArchive crxInArchive = archiveHelper.readArchive(crxFile);
67          assertNotNull(crxInArchive);
68          assertArrayEquals(DummyArchive.PUBLIC_KEY, crxInArchive.getPublicKey());
69          assertArrayEquals(DummyArchive.SIGNATURE, crxInArchive.getSignature());
70          assertArrayEquals(DummyArchive.DATA, crxInArchive.getData());
71      }
